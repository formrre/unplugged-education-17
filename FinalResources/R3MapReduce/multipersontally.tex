\newcommand\PythTr[7][text=black]{%
\coordinate [] (#2) at #5;
\coordinate [] (#4) at #6;
\coordinate [] (#3) at #7;

\draw
  (#2) --
  node[auto] [below,sloped]{\wordsPick} (#3) --
  node[auto] [above,sloped]{\rotatebox{180}{\wordsPick}} 
  node[below,sloped]{\wordsPick} (#4) --
  node[auto] [below,sloped]{\wordsPick}
  (#2);

\coordinate (aux1) at
  ($ (#2) ! {sin(90)} ! 90:(#3) $);
\coordinate (aux2) at
  ($ (aux1) ! {sin(90)} ! 90:(#2) $);

\coordinate (aux3) at
  ($ (#2) ! {sin(90)} ! -90:(#4) $);
\coordinate (aux4) at
  ($ (aux3) ! {sin(90)} ! -90:(#2) $);

\coordinate (aux5) at
  ($ (#4) ! {sin(90)} ! -90:(#3) $);
\coordinate (aux6) at
  ($ (aux5) ! {sin(90)} ! -90:(#4) $);

\begin{scope}[#1]
\draw
  (#2) --
  (aux1) node[midway,auto,swap] {} --
  (aux2) node[midway,auto,swap] {} --
	(#3) node[midway,auto,swap] {};
\draw
  (#2) --
  (aux3) node[midway,auto] {} --
  (aux4) node[midway,auto] {}  --
  (#4) node[midway,auto] {};
\draw
  (#4) --
  (aux5) node[midway,auto] {} --
  (aux6) node[midway,auto] {} --
  (#3) node[midway,auto] {};
\end{scope}
}
